program factorial ;
function fact ( int x ) : int {
        int result = 1 ;
        if [ $x > 1 ] {
                while [ $x > 1 ] {
                        $result = $result * $x ;
                        $x = $x - 1 ;
                }
        }
        ifelse [ $x < 0 ] {
                $result = -1 ;
        }
        return $result ;
}

start
{
        int n = 5 ;
        fact ( $n ) ;
} end
