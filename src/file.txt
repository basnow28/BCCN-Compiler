program factorial ;
function fact ( int n ) : int {
        int result = 1 ;
        if [ $n > 1 ] {
                while [ $n > 1 ] {
                        $result = $result * $n ;
                        $n = $n - 1 ;
                }
        }
        ifelse [ $n < 0 ] {
                $result = -1 ;
        }
        return $result ;
}

start
{
        int n = 5 ;
        fact ( $n ) ;
} end
